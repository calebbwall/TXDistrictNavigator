The correct rule set

For each selected district:
	1.	If the tap/draw point is inside the district polygon → use it directly (this will be very close to where you tapped).
	2.	If it’s outside (edge cases, imprecise hit testing, multipolygons, etc.) → compute the nearest point on the district boundary to the tap, then nudge slightly inward so it becomes inside.

No anchor needed except as a fallback direction for the “nudge inward” step.

Implementation details (do this exactly)

A) Utility: convert {lat,lng} ↔ GeoJSON point

import { point } from "@turf/helpers";

type LatLng = { latitude: number; longitude: number };

function toTurfPoint(ll: LatLng) {
  return point([ll.longitude, ll.latitude]);
}

function fromTurfPoint(p: any): LatLng {
  const [lng, lat] = p.geometry.coordinates;
  return { latitude: lat, longitude: lng };
}

B) Utility: closestPointInsidePolygon(origin, feature)

import booleanPointInPolygon from "@turf/boolean-point-in-polygon";
import polygonToLine from "@turf/polygon-to-line";
import nearestPointOnLine from "@turf/nearest-point-on-line";
import centroid from "@turf/centroid";

function stepToward(a: LatLng, b: LatLng, step: number): LatLng {
  // move from a toward b by "step" degrees (tiny)
  const dx = b.longitude - a.longitude;
  const dy = b.latitude - a.latitude;
  const len = Math.sqrt(dx*dx + dy*dy) || 1;
  return {
    latitude: a.latitude + (dy / len) * step,
    longitude: a.longitude + (dx / len) * step,
  };
}

/**
 * Returns a point that is inside the polygon and as close as possible to origin.
 * If origin is already inside, returns origin.
 */
export function closestPointInsidePolygon(
  origin: LatLng,
  feature: any,
  cachedBoundaryLine: any | null,
): LatLng {
  const originPt = toTurfPoint(origin);

  // 1) If inside, use origin directly.
  if (booleanPointInPolygon(originPt, feature)) return origin;

  // 2) Nearest point on boundary.
  const boundaryLine = cachedBoundaryLine ?? polygonToLine(feature);
  const nearest = nearestPointOnLine(boundaryLine, originPt);
  let candidate = fromTurfPoint(nearest);

  // 3) Nudge inward.
  // Use centroid as an interior direction (only for direction, not placement).
  const c = centroid(feature);
  const centroidLL = fromTurfPoint(c);

  // Nudge amount: tiny degrees. We'll iterate until inside or max tries.
  // (0.00015 deg ~ ~15-20 meters depending on latitude; adjust if needed)
  const step = 0.00015;

  // Try stepping inward multiple times.
  for (let i = 0; i < 25; i++) {
    candidate = stepToward(candidate, centroidLL, step);
    if (booleanPointInPolygon(toTurfPoint(candidate), feature)) return candidate;
  }

  // Fallback: if something is weird with geometry, return centroid as last resort.
  return centroidLL;
}
