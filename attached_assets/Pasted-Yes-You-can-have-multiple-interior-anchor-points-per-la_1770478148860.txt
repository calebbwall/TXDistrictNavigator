import bbox from "@turf/bbox";
import area from "@turf/area";
import centroid from "@turf/centroid";
import booleanPointInPolygon from "@turf/boolean-point-in-polygon";
import { point } from "@turf/helpers";

type LatLng = { latitude: number; longitude: number };

const anchorCache = new Map<string, LatLng[]>();

function fromCoords([lng, lat]: number[]): LatLng {
  return { latitude: lat, longitude: lng };
}

function toPoint(ll: LatLng) {
  return point([ll.longitude, ll.latitude]);
}

function clamp(min: number, max: number, v: number) {
  return Math.max(min, Math.min(max, v));
}

function getDistrictAnchors(districtId: string, feature: any): LatLng[] {
  const cached = anchorCache.get(districtId);
  if (cached) return cached;

  const a = area(feature); // m^2
  const n = clamp(1, 6, Math.ceil(a / 2.5e9)); // tune threshold as needed

  const c = centroid(feature).geometry.coordinates; // [lng, lat]
  const centroidLL = fromCoords(c);

  const anchors: LatLng[] = [centroidLL];

  if (n === 1) {
    anchorCache.set(districtId, anchors);
    return anchors;
  }

  const [minLng, minLat, maxLng, maxLat] = bbox(feature);

  // Build a deterministic grid; increase resolution with n
  const gridSize = n <= 3 ? 5 : 7;
  const pts: LatLng[] = [];

  for (let i = 0; i < gridSize; i++) {
    for (let j = 0; j < gridSize; j++) {
      const lng = minLng + ((i + 0.5) / gridSize) * (maxLng - minLng);
      const lat = minLat + ((j + 0.5) / gridSize) * (maxLat - minLat);
      const ll = { latitude: lat, longitude: lng };
      if (booleanPointInPolygon(toPoint(ll), feature)) pts.push(ll);
    }
  }

  // Greedy farthest-point selection to spread anchors
  function dist2(p: LatLng, q: LatLng) {
    const dx = p.longitude - q.longitude;
    const dy = p.latitude - q.latitude;
    return dx*dx + dy*dy;
  }

  const selected: LatLng[] = [];
  if (pts.length) {
    // seed with point farthest from centroid
    pts.sort((p, q) => dist2(q, centroidLL) - dist2(p, centroidLL));
    selected.push(pts[0]);

    while (selected.length < (n - 1) && selected.length < pts.length) {
      let best = pts[0];
      let bestScore = -1;
      for (const candidate of pts) {
        let minD = Infinity;
        for (const s of selected) minD = Math.min(minD, dist2(candidate, s));
        if (minD > bestScore) {
          bestScore = minD;
          best = candidate;
        }
      }
      selected.push(best);
    }
  }

  anchors.push(...selected.slice(0, n - 1));
  anchorCache.set(districtId, anchors);
  return anchors;
}
